  <script>
    const LINE_URL = "https://line.me/ti/p/MT44kyhx74";
    const LINE_ID_TEXT = "@yuan056788";
    const AUTO_SECONDS = 3;

    // 倒數自動前往
    const countEl = document.getElementById('count');
    const cancelBtn = document.getElementById('cancel');
    let sec = AUTO_SECONDS, cancelled = false, t=null;
    function tick() {
      countEl.textContent = String(sec);
      if (sec <= 0 && !cancelled) { window.location.href = LINE_URL; return; }
      sec--; t = setTimeout(tick, 1000);
    }
    tick();
    cancelBtn.addEventListener('click', () => {
      cancelled = true; clearTimeout(t);
      cancelBtn.textContent = '已取消'; countEl.textContent = '—';
    });

    // 複製 ID
    const toast = document.getElementById('toast');
    document.getElementById('copyBtn').addEventListener('click', async () => {
      try {
        await navigator.clipboard.writeText(LINE_ID_TEXT.replace(/^\s+|\s+$/g, ''));
        toast.classList.add('show');
        setTimeout(() => toast.classList.remove('show'), 1800);
      } catch (e) {
        alert('複製失敗，請手動選取複製');
      }
    });

    // 年份
